#include <iostream>
using namespace std;

int day[2][13] = { { 0,31,28,31,30,31,30,31,31,30,31,30,31 } ,{ 0,31,29,31,30,31,30,31,31,30,31,30,31 } };

class Date
{
public:
                int year, month, day;
               Date( int y = 1970, int m = 1, int d = 1) :year(y), month( m), day(d ) {};
                friend int getDays(Date & d1, Date &d2);
};
bool isLeapYear(int year)
{
                if ((year % 4 == 0 && year % 100 != 0) || ( year % 400 == 0))
               {
                               return true ;
               }
                else
               {
                               return false ;
               }
}
int countDay(int y, int m, int d, int flag)
{
                int sum = 0;
                int p = 0;
                if (isLeapYear(y ))
               {
                              p = 1;
               }
                if (flag )
               {
                              sum += d;
                               for (int i = 1;i < m;i++)
                              {
                                             sum += day[p][i];
                              }
               }
                else
               {
                              sum += (day[p][ m] - d );
                               for (int i = m + 1;i < 13;i++)
                              {
                                             sum += day[p][i];
                              }
               }
                return sum;
}
int getDays(Date & d1, Date & d2)
{
                int sum = 0;
                if (d1 .year < d2.year)
               {
                               for (int i = d1.year + 1;i < d2.year;i++)
                              {
                                              if (isLeapYear(i))
                                             {
                                                            sum += 366;
                                             }
                                              else
                                             {
                                                            sum += 365;
                                             }
                              }
                              sum += countDay( d1.year, d1 .month, d1.day, 0);
                              sum += countDay( d2.year, d2 .month, d2.day, 1);
                               return sum;
               }
                if(d1 .year==d2.year)
               {
                               int day1= countDay(d1 .year, d1.month, d1.day, 1);
                               int day2= countDay(d2 .year, d2.month, d2.day, 1);
                              sum = day2 - day1;
                               return sum;
               }
}
